# 冒泡排序

- (void)bubbleSort:(NSArray *)array {
    NSMutableArray *mArray = [NSMutableArray arrayWithArray:array];
    for (int i = 0; i < mArray.count - 1; i ++) {
        for (int j = i + 1; j < mArray.count; j ++) {// j = i + 1，直接从下一位开始遍历
            if ([mArray[i] intValue] > [mArray[j] intValue]) {
                // 将比mArray[i]小的数 与mArray[i]进行交换
                id temp = mArray[j];
                mArray[j] = mArray[i];
                mArray[i] = temp;
            }
        }
    }
    NSLog(@"mArray--->%@", mArray);
}

# 选择排序

- (void)selectSort:(NSArray *)array {
    NSMutableArray *mArray = [NSMutableArray arrayWithArray:array];
    for (int i = 0; i < mArray.count - 1; i ++) {
        int maxNumIndex = 0;
        for (int j = 0; j < mArray.count - i; j ++) {
            if ([mArray[maxNumIndex] intValue] < [mArray[j] intValue]) {
                maxNumIndex = j;
            }
        }
        id temp = mArray[array.count - i - 1];// 记录数组最后一位
        mArray[array.count - i - 1] = mArray[maxNumIndex];// 将最大值移到最后一位
        mArray[maxNumIndex] = temp;// 将最后一位的值移到 最大值原来的索引
    }
    NSLog(@"mArray--->%@", mArray);
}

# 快速排序

- (void)quickSort:(NSArray *)array {
    NSMutableArray *mArray = [NSMutableArray arrayWithArray:array];
    [self quickSort:mArray start:0 end:(int)(mArray.count-1)];
    NSLog(@"mArray--->%@", mArray);
}

- (void)quickSort:(NSMutableArray *)array start:(int)start end:(int)end {
    if (start > end) return;
    int i, j, base;
    i = start;
    j = end;
    base = [array[start] intValue];
    while (i < j) {
        while (i < j && [array[j] intValue] >= base) {
            j--;
        }
        while (i < j && [array[i] intValue] <= base) {
            i++;
        }
        if (i < j) {
            [self swap:array left:i right:j];
        }
    }
    [self swap:array left:start right:i];
    [self quickSort:array start:start end:j - 1];
    [self quickSort:array start:j + 1 end:end];
}

- (void)swap:(NSMutableArray *)array left:(int)left right:(int)right {
    id temp = array[left];
    array[left] = array[right];
    array[right] = temp;
}
